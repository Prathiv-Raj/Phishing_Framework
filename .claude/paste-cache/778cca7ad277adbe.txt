version: '3.8'

# Data flow:
#   gophish writes  → /opt/ab-phish/data/gophish.db  (bind-mounted from ./Gophish/data)
#   dashboard reads → /data/gophish/gophish.db        (same host dir, different mount point)
#   training reads  → /data/gophish/gophish.db        (same host dir, different mount point)
#   training writes → training-data volume (training_status.db)
#   dashboard reads → training-data volume (training_status.db)
#   dashboard writes→ dashboard-data volume (auth.db)

services:
  # ─── AB-PHISH (GOPHISH) ────────────────────────────────────────────────────
  gophish:
    image: cyberbothub/phishing:abphish
    container_name: ab-phish
    ports:
      - "3333:3333"   # Admin panel  → https://localhost:3333
      - "8080:8080"   # Phishing server → http://localhost:8080
    volumes:
      - ./abphish/data:/opt/ab-phish/data   # gophish.db stored at /opt/ab-phish/data/gophish.db
    restart: unless-stopped
    networks:
      - phishing-net

  # ─── DASHBOARD ─────────────────────────────────────────────────────────────
  dashboard:
    image: cyberbothub/phishing:dashboard
    container_name: dashboard
    ports:
      - "5050:5050"
    environment:
      - GOPHISH_DB_PATH=/data/gophish/gophish.db
      - AUTH_DB_PATH=/data/dashboard/auth.db
      - TRAINING_DB_PATH=/data/training/training_status.db
      - PORT=5050
      - SECRET_KEY=d705a5883e604d4be669caa74829e8a8104b63698d0cefcb956ab8853e0d79e4
    volumes:
      - ./Gophish/data:/data/gophish         # shared gophish DB (read)
      - dashboard-data:/data/dashboard        # auth.db (read/write)
      - training-data:/data/training          # training_status.db (read)
    depends_on:
      - gophish
    restart: unless-stopped
    networks:
      - phishing-net

  # ─── TRAINING ──────────────────────────────────────────────────────────────
  training:
    image: cyberbothub/phishing:training
    container_name: training
    ports:
      - "9090:9090"
    environment:
      - GOPHISH_DB_PATH=/data/gophish/gophish.db
      - TRAINING_DB_PATH=/data/training/training_status.db
      - PORT=9090
    volumes:
      - ./Gophish/data:/data/gophish         # shared gophish DB (read)
      - training-data:/data/training          # training_status.db (read/write)
    depends_on:
      - gophish
    restart: unless-stopped
    networks:
      - phishing-net

networks:
  phishing-net:
    driver: bridge

volumes:
  dashboard-data:    # auth.db for dashboard login
  training-data:     # training_status.db shared by training + dashboard
